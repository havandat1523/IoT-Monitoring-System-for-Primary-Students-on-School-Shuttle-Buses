    setTimeout(function() {
        var announcements = [

            "Thực đơn bán trú hôm nay có món gà rán và canh súp lơ. Đảm bảo các con sẽ rất thích!",

            "Lưu ý an toàn: Khu vực cổng trường cấm đỗ xe ô tô quá 5 phút để tránh ùn tắc vào giờ tan tầm.",
            "Tin vui: Đội tuyển bơi lội của trường vừa xuất sắc giành giải Nhất cấp thành phố. Xin chúc mừng!",
            "Chương trình ngoại khóa 'Khám phá thiên nhiên' sẽ diễn ra vào Thứ Sáu. Hạn chót nộp phiếu là ngày mai.",
            "Phòng y tế nhắc nhở: Đang vào mùa dịch, các con vui lòng mặc quần áo dài tay và thoa kem chống muỗi.",
            "Lời khen ngợi: Tập thể lớp 3A hôm nay đã có hành động rất đẹp khi nhặt được của rơi trả lại người mất."
        ];

        var currentIndex = 0;
        var tickerElement = document.getElementById("tickerContent");

        // Nếu tìm thấy khung HTML thì mới bắt đầu chạy
        if(tickerElement) {
            // Hiện câu đầu tiên
            tickerElement.innerText = announcements[0];
            tickerElement.classList.add("fade-in");

            // Đặt lịch đổi câu sau mỗi 3.5 giây
            setInterval(function() {
                // Làm mờ và tụt xuống
                tickerElement.classList.remove("fade-in");
                tickerElement.classList.add("fade-out");

                // Đợi 0.4s rồi thay chữ và trượt lên
                setTimeout(function() {
                    currentIndex = (currentIndex + 1) % announcements.length;
                    tickerElement.innerText = announcements[currentIndex];
                    
                    tickerElement.classList.remove("fade-out");
                    tickerElement.classList.add("fade-in");
                }, 400); 
            }, 3500);
        }
    }, 500);
