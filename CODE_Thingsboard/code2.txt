        <input type="date" id="leaveDate" class="input-field">
        
        <button id="btnSubmit" class="btn-submit">Gửi thông tin nghỉ học</button>

        <h3 style="color: #27ae60;">Đăng ký thành công!</h3>
        <p style="color: #555; line-height: 1.5; margin-bottom: 25px;">
            Phụ huynh nên gọi điện trực tiếp cho <b>Giáo viên chủ nhiệm</b> hoặc <b>Tài xế</b> để xác nhận thêm nhé!
        </p>
        <button id="btnOkAlert" class="btn-submit alert-btn">OK, Tôi hiểu</button>
    </div>
</div>

<script>
    var modal = document.getElementById("formModal");
    var alertModal = document.getElementById("alertModal"); // Khai báo modal mới
    
    var btnOpen = document.getElementById("openModalBtn");
    var btnClose = document.getElementById("closeModalBtn");
    var btnSubmit = document.getElementById("btnSubmit");
    var btnOkAlert = document.getElementById("btnOkAlert"); // Nút OK của modal mới
    var statusMsg = document.getElementById("statusMsg");

    // Mở form đăng ký
    btnOpen.onclick = function() {
        modal.style.display = "flex";
        statusMsg.innerText = ""; 
        btnSubmit.innerText = "Gửi thông tin nghỉ học";
        btnSubmit.disabled = false;
        btnSubmit.style.background = "#27ae60"; 
    }

    // Nút tắt form
    btnClose.onclick = function() { 
        modal.style.display = "none"; 
    }

    // Khi phụ huynh bấm "OK, Tôi hiểu" trên hộp thông báo mới
    btnOkAlert.onclick = function() {
        alertModal.style.display = "none"; // Tắt hộp thông báo
        
        // Dọn sạch form nhập liệu để người sau dùng
        document.getElementById("studentId").value = "";
        document.getElementById("studentName").value = "";
        document.getElementById("leaveDate").value = "";
    }

    // Xử lý Gửi dữ liệu
    btnSubmit.onclick = function() {
        var id = document.getElementById("studentId").value.trim();
        var name = document.getElementById("studentName").value.trim();
        var rawDate = document.getElementById("leaveDate").value;

        if(!id || !name || !rawDate) {
            statusMsg.innerText = "⚠️ Ối! Bạn nhập thiếu thông tin rồi.";
            statusMsg.style.color = "#e74c3c";
            return;
        }

        var dateParts = rawDate.split("-");
        var formattedDate = dateParts[2] + "/" + dateParts[1] + "/" + dateParts[0];

        btnSubmit.disabled = true;
        btnSubmit.innerHTML = "<span class='loading-dots'>Đang ghi vào sổ</span>";
        statusMsg.innerText = "";

        // ========================================================
        // NHỚ DÁN LẠI ĐƯỜNG LINK GOOGLE SCRIPT
        var scriptURL = "https://script.google.com/macros/s/AKfycbxw0pxOAQZ3R42Ly-519VM4gEww0pZO-2QpeFbzoS9xIMec2oi8g6iZL0CadGs5wAhJ/exec";
        // ========================================================

        var finalURL = scriptURL + "?action=leave&id=" + encodeURIComponent(id) + "&name=" + encodeURIComponent(name) + "&date=" + encodeURIComponent(formattedDate);

        fetch(finalURL)
        .then(response => response.text())
        .then(result => {
            if(result === "OK") {
                btnSubmit.style.background = "#27ae60";
                btnSubmit.innerHTML = "✔ Đã ghi thành công!";
                statusMsg.innerText = "Chuẩn bị tải hộp thoại...";
                statusMsg.style.color = "#27ae60";
                
                // ĐỢI 1 GIÂY RỒI TẮT FORM, BẬT HỘP THÔNG BÁO "GỌI ĐIỆN" LÊN
                setTimeout(function(){ 
                    modal.style.display = "none";     // Tắt form nhập liệu
                    alertModal.style.display = "flex"; // Bật bảng nhắc nhở gọi điện
                }, 1000); 
            } 
            else if (result === "NOT_FOUND") {
                btnSubmit.disabled = false;
                btnSubmit.innerHTML = "Thử lại";
                statusMsg.innerText = "❌ Lỗi: Không tìm thấy Mã học sinh trong danh sách!";
                statusMsg.style.color = "#e74c3c";
            } 
            else {
                btnSubmit.disabled = false;
                btnSubmit.innerHTML = "Thử lại";
                statusMsg.innerText = "❌ Lỗi không xác định: " + result;
                statusMsg.style.color = "#e74c3c";
            }
        })
        .catch(error => {
            btnSubmit.disabled = false;
            btnSubmit.innerHTML = "Thử lại";
            statusMsg.innerText = "❌ Lỗi: Không thể kết nối với Google Sheets!";
            statusMsg.style.color = "#e74c3c";
        });
    }
</script>
